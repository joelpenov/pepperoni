def default_values(apps,schema_editor):
    UnitOfMeasure = apps.get_model("inventory", "UnitOfMeasure")
    Product = apps.get_model("inventory", "Product")
    unit_of_measure = UnitOfMeasure.objects.create(description='Unidad', abbreviation='Und')
    unit_of_measure.save()

    for product in Product.objects.all():
        product.unit_of_measure =unit_of_measure
        product.save()
